#version 330

in vec2 pass_textureCoords;

out vec4 out_colour;

uniform vec3 colour;
uniform float alpha;
uniform sampler2D fontAtlas;
uniform int yMin;
uniform int yMax;

void main(void){

    if (gl_FragCoord.y < yMin || gl_FragCoord.y > yMax) {
        discard;
    }

	out_colour = vec4(colour, texture(fontAtlas, pass_textureCoords).a * alpha);

}
